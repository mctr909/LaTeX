[ ラプラス変換 ] \begin{align} \mathscr{L}\left[f(t)\right] &:= \int^{\infty}_{0}f(t)e^{-st}dt \\ \end{align}
\begin{align} \mathscr{L}\left[e^{\alpha t}\right]&=\int^{\infty}_{0}{e^{\alpha t}}e^{-st}dt \\ &=\int^{\infty}_{0}{e^{(\alpha -s)t}}dt \\ &=-\cfrac{1}{s-\alpha}\left[e^{(\alpha-s)t}\right]^{\infty}_{0} \\ &=-\cfrac{1}{s-\alpha}\left(\lim_{t \to \infty}{e^{(\alpha-s)t}}-e^{(\alpha-s)\cdot0}\right) \\ &=-\cfrac{1}{s-\alpha}\left(0-1\right)\Leftarrow\mathrm{Re}\left(s\right) \gt \mathrm{Re}\left(\alpha\right) \\ &=\cfrac{1}{s-\alpha} \end{align}
\begin{align} \mathscr{L}\left[\cos{\omega t}\right]&=\int^{\infty}_{0}{\cos(\omega t)}e^{-st}dt \\ &=\int^{\infty}_{0}{\cfrac{\left(e^{i\omega t}+e^{-i\omega t}\right)}{2}e^{-st}}dt \\ &=\frac{1}{2}\left(\int^{\infty}_{0}{e^{i\omega t}e^{-st}}dt + \int^{\infty}_{0}{e^{-i\omega t}e^{-st}}dt\right) \\ &=\frac{1}{2}\left(\mathscr{L}\left[e^{i\omega t}\right] + \mathscr{L}\left[e^{-i\omega t}\right]\right) \quad\cdots i\omega,-i\omegaを\mathscr{L}\left[e^{\alpha t}\right]の\alphaに代入 \\ &=\frac{1}{2}\left(\cfrac{1}{s-i\omega} + \cfrac{1}{s+i\omega}\right) \\ &=\frac{1}{2}\left(\cfrac{s+i\omega}{(s-i\omega)(s+i\omega)} + \cfrac{s-i\omega}{(s+i\omega)(s-i\omega)}\right) \\ &=\frac{1}{2}\left(\cfrac{s+i\omega}{s^2+\omega^2} + \cfrac{s-i\omega}{s^2+\omega^2}\right) \\ &=\frac{1}{2}\left(\cfrac{2s}{s^2+\omega^2}\right) \\ &=\cfrac{s}{s^2+\omega^2} \end{align}
\begin{align} \mathscr{L}\left[\sin{\omega t}\right]&=\int^{\infty}_{0}{\sin(\omega t)}e^{-st}dt \\ &=\int^{\infty}_{0}{\cfrac{\left(e^{i\omega t}-e^{-i\omega t}\right)}{2i}e^{-st}}dt \\ &=\frac{1}{2i}\left(\int^{\infty}_{0}{e^{i\omega t}e^{-st}}dt - \int^{\infty}_{0}{e^{-i\omega t}e^{-st}}dt\right) \\ &=\frac{1}{2i}\left(\mathscr{L}\left[e^{i\omega t}\right] - \mathscr{L}\left[e^{-i\omega t}\right]\right) \quad\cdots i\omega,-i\omegaを\mathscr{L}\left[e^{\alpha t}\right]の\alphaに代入 \\ &=\frac{1}{2i}\left(\cfrac{1}{s-i\omega} - \cfrac{1}{s+i\omega}\right) \\ &=\frac{1}{2i}\left(\cfrac{s+i\omega}{(s-i\omega)(s+i\omega)} - \cfrac{s-i\omega}{(s+i\omega)(s-i\omega)}\right) \\ &=\frac{1}{2i}\left(\cfrac{s+i\omega}{s^2+\omega^2} - \cfrac{s-i\omega}{s^2+\omega^2}\right) \\ &=\frac{1}{2i}\left(\cfrac{2i\omega}{s^2+\omega^2}\right) \\ &=\cfrac{\omega}{s^2+\omega^2} \end{align}
