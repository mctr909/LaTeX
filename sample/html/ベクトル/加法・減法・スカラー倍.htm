\begin{align} \boldsymbol{a}+\boldsymbol{b}:=\begin{pmatrix} \boldsymbol{a}_1+\boldsymbol{b}_1 \\ \boldsymbol{a}_2+\boldsymbol{b}_2 \\ \vdots \\ \boldsymbol{a}_n+\boldsymbol{b}_n \end{pmatrix} \end{align}
\begin{align} \boldsymbol{a}-\boldsymbol{b}:=\begin{pmatrix} \boldsymbol{a}_1-\boldsymbol{b}_1 \\ \boldsymbol{a}_2-\boldsymbol{b}_2 \\ \vdots \\ \boldsymbol{a}_n-\boldsymbol{b}_n \end{pmatrix} \end{align}
\begin{align} k\boldsymbol{a} := \begin{pmatrix} k{\boldsymbol{a}_1} \\ k{\boldsymbol{a}_2} \\ \vdots \\ k{\boldsymbol{a}_n} \end{pmatrix} \end{align}
\begin{align} \frac{\boldsymbol{a}}{k} := \begin{pmatrix} \cfrac{\boldsymbol{a}_1}{k} \\ \cfrac{\boldsymbol{a}_2}{k} \\ \vdots \\ \cfrac{\boldsymbol{a}_n}{k} \end{pmatrix} \end{align}
