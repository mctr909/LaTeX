[ 2乗の公式 ]
\begin{align} 三平方の定理の両辺を斜辺で割ると \\ 三角関数の2乗の公式が得られる。 \end{align} \begin{align} r^2 &= x^2 + y^2 \\ &= \left(r\cos{\theta}\right)^2 + \left(r\sin{\theta}\right)^2 \\ \therefore 1 &= \cfrac{\left(r\cos{\theta}\right)^2}{r^2} + \cfrac{\left(r\sin{\theta}\right)^2}{r^2} \\ &= \cos^2{\theta}+\sin^2{\theta} \end{align} \begin{align} 変形して\cos,\sin,\tanについてまとめると \end{align} \begin{align} \cos^2{\theta} &= 1 - \sin^2{\theta} \\ \sin^2{\theta} &= 1 - \cos^2{\theta} \\ \tan^2{\theta} &= \frac{\sin^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1-\cos^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1}{\cos^2{\theta}} - 1 \\ \cfrac{1}{\tan^2{\theta}} &= \frac{\cos^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1-\sin^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1}{\sin^2{\theta}} - 1 \end{align}
[ 周期性と対称性 ]
\begin{align} \cos\left(\pm\theta+2n{\pi}\right) &= \cos\theta \\ \sin\left(\pm\theta+2n{\pi}\right) &= \pm\sin\theta \end{align} \begin{align} \cos\left(\pm\theta+{\left(2n+1\right)\pi}\right) &= -\cos\theta \\ \sin\left(\pm\theta+{\left(2n+1\right)\pi}\right) &= \mp\sin\theta \end{align} \begin{align} \tan\left(\pm\theta+n{\pi}\right) = \pm\tan\theta \end{align}
[ 余角の公式 ]
\begin{align} \cos\left(\pm\cfrac{\pi}{2}-\theta\right) &= \pm\sin\theta \\ \sin\left(\pm\cfrac{\pi}{2}-\theta\right) &= \pm\cos\theta \\ \tan\left(\pm\cfrac{\pi}{2}-\theta\right) &= \cfrac{\sin\left(\pm\cfrac{\pi}{2}-\theta\right)}{\cos\left(\pm\cfrac{\pi}{2}-\theta\right)} \\ = \cfrac{\pm\cos\theta}{\pm\sin\theta} &= \cfrac{1}{\tan\theta} \end{align}
\begin{align} \cos\left(\theta\pm\cfrac{\pi}{2}\right) &= \mp\sin\theta \\ \sin\left(\theta\pm\cfrac{\pi}{2}\right) &= \pm\cos\theta \\ \tan\left(\theta\pm\cfrac{\pi}{2}\right) &= \cfrac{\sin\left(\theta\pm\cfrac{\pi}{2}\right)}{\cos\left(\theta\pm\cfrac{\pi}{2}\right)} \\ = \cfrac{\pm\cos\theta}{\mp\sin\theta} &= -\cfrac{1}{\tan\theta} \end{align}
