\begin{align} {\small複号同順でまとめると} \end{align} \begin{align} \cos\left(\pm\theta+\frac{\pi}{2}\right)&=\mp\sin\theta \\ \sin\left(\pm\theta+\frac{\pi}{2}\right)&=\cos\theta \end{align}
\begin{align} {\small複号同順でまとめると} \end{align} \begin{align} \cos\left(\pm\theta-\frac{\pi}{2}\right)&=\pm\sin\theta \\ \sin\left(\pm\theta-\frac{\pi}{2}\right)&=-\cos\theta \end{align}
