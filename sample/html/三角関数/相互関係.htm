[ 余角の公式 ] [ 平方関係 ]
\begin{align} 三平方の定理の両辺を斜辺で割ると \\ 三角関数の平方関係が得られる。 \end{align} \begin{align} r^2 &= x^2 + y^2 \\ &= \left(r\cos{\theta}\right)^2 + \left(r\sin{\theta}\right)^2 \\ \therefore 1 &= \cfrac{\left(r\cos{\theta}\right)^2}{r^2} + \cfrac{\left(r\sin{\theta}\right)^2}{r^2} \\ &= \cos^2{\theta}+\sin^2{\theta} \end{align} \begin{align} 変形して\cos^2,\sin^2,\tan^2についてまとめると、 \end{align} \begin{align} \cos^2{\theta} &= 1 - \sin^2{\theta} \\ \sin^2{\theta} &= 1 - \cos^2{\theta} \\ \tan^2{\theta} &= \frac{\sin^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1-\cos^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1}{\cos^2{\theta}} - 1 \\ \\ \frac{1}{\cos^2{\theta}} &= \tan^2{\theta} + 1 \\ \cfrac{1}{\tan^2{\theta}} &= \frac{\cos^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1-\sin^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1}{\sin^2{\theta}} - 1 \end{align}
\begin{align} 複号同順でまとめると \end{align} \begin{align} \cos\left(\pm\cfrac{\pi}{2}-\theta\right)&=\pm\sin\theta \\ \sin\left(\pm\cfrac{\pi}{2}-\theta\right)&=\pm\cos\theta \end{align}
\begin{align} 複号同順でまとめると \end{align} \begin{align} \cos\left(\theta\pm\cfrac{\pi}{2}\right)&=\mp\sin\theta \\ \sin\left(\theta\pm\cfrac{\pi}{2}\right)&=\pm\cos\theta \end{align}
\begin{align} &\left(\pm\cfrac{\pi}{2}-\theta\right)と\left(\theta\pm\cfrac{\pi}{2}\right)について \\ &\thetaの複号を{\Large\pm_t}\ \ \pm\cfrac{\pi}{2}の複号を{\Large\pm_\phi}\ として複号同順でまとめると \end{align} \begin{align} \cos\left({\pm_t}\theta{\pm_\phi}\cfrac{\pi}{2}\right)&=\begin{cases} {\mp_\phi}\sin\theta &{\pm_t}\cdots正\\ {\pm_\phi}\sin\theta &{\pm_t}\cdots負 \end{cases} \\ \sin\left({\pm_t}\theta{\pm_\phi}\cfrac{\pi}{2}\right)&=\begin{cases} {\pm_\phi}\cos\theta &{\pm_t}\cdots正負によらない \end{cases} \\ \tan\left({\pm_t}\theta{\pm_\phi}\cfrac{\pi}{2}\right)&=\begin{cases} -\cfrac{1}{\tan\theta} &{\pm_t}\cdots正\\ \cfrac{1}{\tan\theta} &{\pm_t}\cdots負 \end{cases} \\ &\because\frac{\sin\left({\pm_t}\theta{\pm_\phi}\cfrac{\pi}{2}\right)}{\cos\left({\pm_t}\theta{\pm_\phi}\cfrac{\pi}{2}\right)}=\begin{cases} \cfrac{{\pm_\phi}\cos\theta}{{\mp_\phi}\sin\theta} &{\pm_t}\cdots正\\ \cfrac{{\pm_\phi}\cos\theta}{{\pm_\phi}\sin\theta} &{\pm_t}\cdots負 \end{cases} \end{align}
