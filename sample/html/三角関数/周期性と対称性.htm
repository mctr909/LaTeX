[ 周期性 ]
\begin{align} \cos{\smallと}\sin{\smallの周期は}2\pi \end{align} \begin{align} &角度の差が{\Large2n\pi}なら同じ角度として扱える \\ &{\Large1}周期の間に同じ値が存在する \end{align} \begin{align} \tan{\smallの周期は}\pi \end{align} \begin{align} 角度の差が{\Large n\pi}なら同じ値になる \end{align}
\begin{align} {\large\theta}{を}{\large0\le\phi\lt2\pi}{の範囲に収める場合} \end{align} \begin{align} {\large n}&= {\large\frac{\theta}{2\pi}\cdots}{整数部分を得る} \\ {\large\phi} &= {\large\theta - 2n\pi} \end{align} \begin{align} {\large\theta}{を}{\large -\pi\le\phi\lt\pi}{の範囲に収める場合} \end{align} \begin{align} {\large n}&= {\large\frac{\theta}{\pi}\cdots}{整数部分を得る} \\ {\large\phi} &= \begin{cases} {\large\theta - n\pi} & ({\large n\cdots}{偶数}) \\ {\large\theta - (n+1)\pi} & ({\large n\cdots}{奇数}) \end{cases} \end{align}
