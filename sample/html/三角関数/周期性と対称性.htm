[ 周期性 ]
\begin{align} \cosと\sinの周期は{\large2\pi} \end{align} \begin{align} &角度の差が{\Large2n\pi}なら同じ角度として扱える \\ &{\Large1}周期の間に同じ値が存在する \end{align} \begin{align} \tanの周期は{\large\pi} \end{align} \begin{align} 角度の差が{\Large n\pi}なら同じ値になる \end{align}
\begin{align} {\large\theta}を{0\le\phi\lt2\pi}の範囲に収める場合 \end{align} \begin{align} n &= \cfrac{\theta}{2\pi} \cdots{\small整数部分を得る} \\ \phi &= \theta - 2n\pi \end{align} \begin{align} {\large\theta}を{-\pi\le\phi\lt\pi}の範囲に収める場合 \end{align} \begin{align} n &= \cfrac{\theta}{2\pi} \cdots{\small整数部分を得る} \\ \alpha &= \theta - 2n\pi \\ \phi &= \begin{cases} \alpha & (\alpha\lt\pi) \\ \alpha - 2\pi & (\alpha\ge\pi) \end{cases} \end{align}
[ 対称性 ]
\begin{align} 複号同順でまとめると \end{align} \begin{align} \cos\left(\pm\theta+n\pi\right) &= \begin{cases} \quad\cos\theta&(n{\cdots}{\small偶数}) \\ -\cos\theta&(n{\cdots}{\small奇数}) \end{cases} \ (\pm\cdots{\small正負によらない})\\ \sin\left(\pm\theta+n\pi\right) &= \begin{cases} \pm\sin\theta&(n{\cdots}{\small偶数}) \\ \mp\sin\theta&(n{\cdots}{\small奇数}) \end{cases} \\ \tan\left(\pm\theta+n\pi\right) &= \pm\tan\theta \end{align}
