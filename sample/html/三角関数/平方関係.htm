[ 平方関係 ]
\begin{align} {\small三平方の定理の両辺を斜辺で割ると} \\ {\small三角関数の平方関係が得られる。} \end{align} \begin{align} r^2 &= x^2 + y^2 \\ &= \left(r\cos{\theta}\right)^2 + \left(r\sin{\theta}\right)^2 \\ \therefore 1 &= \cos^2{\theta}+\sin^2{\theta} \end{align} \begin{align} {\small変形して}\cos^2,\sin^2,\tan^2,\frac{1}{\cos^2},\frac{1}{\tan^2}{\smallを得る} \end{align} \begin{align} \cos^2{\theta} &= 1 - \sin^2{\theta} \\ \sin^2{\theta} &= 1 - \cos^2{\theta} \\ \tan^2{\theta} &= \frac{\sin^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1-\cos^2{\theta}}{\cos^2{\theta}} \\ &= \frac{1}{\cos^2{\theta}} - 1 \\ \frac{1}{\cos^2{\theta}} &= \tan^2{\theta} + 1 \\ \frac{1}{\tan^2{\theta}} &= \frac{\cos^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1-\sin^2{\theta}}{\sin^2{\theta}} \\ &= \frac{1}{\sin^2{\theta}} - 1 \end{align}
