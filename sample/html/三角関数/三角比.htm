[ 直角三角形　各辺の名称 ]
　　　
[ 三角比 ]
\begin{align} \cos{\theta} &:= \cfrac{隣辺}{斜辺} = \frac{x}{r} \\ \sin{\theta} &:= \cfrac{対辺}{斜辺} = \frac{y}{r} \\ \tan{\theta} &:= \cfrac{対辺}{隣辺} = \frac{y}{x} \\ &= \cfrac{\cfrac{y}{r}}{\cfrac{x}{r}} = \cfrac{\sin{\theta}}{\cos{\theta}} \end{align}
[ 弧度法と度数法の関係 ] \begin{align} \mathrm{rad} &= \mathrm{deg}\frac{\pi}{180}\\ \mathrm{deg} &= \mathrm{rad}\frac{180}{\pi} \end{align} \begin{align} \mathrm{rad}(ラジアン) &= 弧度\\ \mathrm{deg}(ディグリー) &= 度数 \end{align}
