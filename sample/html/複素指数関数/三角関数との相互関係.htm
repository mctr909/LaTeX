[ cosθを取得 ] \begin{align} e^{i\theta} + e^{-i\theta} &=(\cos{\theta} + i\sin{\theta}) + (\cos{\theta} - i\sin{\theta}) \\ &=\cos{\theta} + \cos{\theta} + i\sin{\theta} - i\sin{\theta} \\ &=2\cos{\theta} \\ &\therefore \cos{\theta} = \frac{e^{i\theta} + e^{-i\theta}}{2} \end{align}
[ sinθを取得 ] \begin{align} e^{i\theta} - e^{-i\theta} &=(\cos{\theta} + i\sin{\theta}) - (\cos{\theta} - i\sin{\theta}) \\ &=\cos{\theta} - \cos{\theta} + i\sin{\theta} + i\sin{\theta} \\ &=2i\sin{\theta} \\ &\therefore \sin{\theta} = \frac{e^{i\theta} - e^{-i\theta}}{2i} \end{align}
[ tanθを取得 ] \begin{align} \tan{\theta} &= \frac{\sin{\theta}}{\cos{\theta}} \\ &=\cfrac{\cfrac{e^{i\theta} - e^{-i\theta}}{2i}}{\cfrac{e^{i\theta} + e^{-i\theta}}{2}} \quad...\cos{\theta}と\sin{\theta}の取得方法を使う \\ &=\frac{e^{i\theta} - e^{-i\theta}}{i(e^{i\theta} + e^{-i\theta})} \cdot \frac{i}{i} \quad...分母の純虚数を有理化 \\ &=-i\frac{e^{i\theta} - e^{-i\theta}}{e^{i\theta}+e^{-i\theta}} \\ &=i\frac{e^{-i\theta} - e^{i\theta}}{e^{-i\theta} + e^{i\theta}} \end{align}
