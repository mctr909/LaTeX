[ 偏角の差 ] \begin{align} e^{i(\alpha-\beta)} &= e^{i\alpha}e^{-i\beta} \\ &= (\cos{\alpha} + i\sin{\alpha}) (\cos{\beta} - i\sin{\beta}) \\ &= \cos{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot i\sin{\beta} + i\sin{\alpha} \cdot \cos{\beta} - i\sin{\alpha} \cdot i\sin{\beta} \\ &= \cos{\alpha} \cdot \cos{\beta} + \sin{\alpha} \cdot \sin{\beta} + i(\sin{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot \sin{\beta}) \end{align}
\begin{align} \cos(\alpha-\beta) &=\frac{e^{i(\alpha-\beta)} + e^{-i(\alpha-\beta)}}{2} \\ &=\frac{e^{i\alpha}e^{-i\beta}}{2} + \frac{e^{-i\alpha}e^{i\beta}}{2}\\ &=\frac{(\cos{\alpha} + i\sin{\alpha})(\cos{\beta} - i\sin{\beta})}{2} \\ &+\frac{(\cos{\alpha} - i\sin{\alpha})(\cos{\beta} + i\sin{\beta})}{2} \\ &=\frac{\cos{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot i\sin{\beta} + i\sin{\alpha} \cdot \cos{\beta} - i\sin{\alpha} \cdot i\sin{\beta} }{2} \\ &+\frac{\cos{\alpha} \cdot \cos{\beta} + \cos{\alpha} \cdot i\sin{\beta} - i\sin{\alpha} \cdot \cos{\beta} - i\sin{\alpha} \cdot i\sin{\beta} }{2} \\ &=\cos{\alpha} \cdot \cos{\beta} + \sin{\alpha} \cdot \sin{\beta} \end{align} \begin{align} 偏角の差の実部を取ると\cos(\alpha-\beta)が得られることがわかる \end{align}
\begin{align} \sin(\alpha-\beta) &= \frac{e^{i(\alpha-\beta)} - e^{-i(\alpha-\beta)}}{2i} \\ &=\frac{e^{i\alpha}e^{-i\beta}}{2i} - \frac{e^{-i\alpha}e^{i\beta}}{2i} \\ &=\frac{(\cos{\alpha} + i\sin{\alpha})(\cos{\beta} - i\sin{\beta})}{2i} \\ &-\frac{(\cos{\alpha} - i\sin{\alpha})(\cos{\beta} + i\sin{\beta})}{2i} \\ &=\frac{\cos{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot i\sin{\beta} + i\sin{\alpha} \cdot \cos{\beta} - i\sin{\alpha} \cdot i\sin{\beta} }{2i} \\ &-\frac{\cos{\alpha} \cdot \cos{\beta} + \cos{\alpha} \cdot i\sin{\beta} - i\sin{\alpha} \cdot \cos{\beta} - i\sin{\alpha} \cdot i\sin{\beta} }{2i} \\ &=\sin{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot \sin{\beta} \end{align} \begin{align} 偏角の差の虚部を取ると\sin(\alpha-\beta)が得られることがわかる \end{align}
