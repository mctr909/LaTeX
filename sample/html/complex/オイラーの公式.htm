\begin{align} e^{i\theta} &= \cos{\theta} + i\sin{\theta} \\ e^{-i\theta} &= \cos{\theta} - i\sin{\theta} \\ \end{align} [ 公式 ]
\begin{align} &e^{i\theta} + e^{-i\theta} \\ &=\left( \cos{\theta} + i\sin{\theta} \right) + \left( \cos{\theta} - i\sin{\theta} \right) \\ &=\cos{\theta} + \cos{\theta} + i\sin{\theta} - i\sin{\theta} \\ &=2\cos{\theta} \\ &\therefore \cos{\theta} = \frac{e^{i\theta} + e^{-i\theta}}{2} \end{align} [ cosθの取得方法 ]
\begin{align} &e^{i\theta} - e^{-i\theta} \\ &=\left( \cos{\theta} + i\sin{\theta} \right) - \left( \cos{\theta} - i\sin{\theta} \right) \\ &=\cos{\theta} - \cos{\theta} + i\sin{\theta} + i\sin{\theta} \\ &=2i\sin{\theta} \\ &\therefore \sin{\theta} = \frac{e^{i\theta} - e^{-i\theta}}{2i} \end{align} [ sinθの取得方法 ]
\begin{align} &\tan{\theta} = \frac{\sin{\theta}}{\cos{\theta}}\\ &=\frac{\frac{e^{i\theta} - e^{-i\theta}}{2i}}{\frac{e^{i\theta} + e^{-i\theta}}{2}}= \frac{e^{i\theta} - e^{-i\theta}}{i\left(e^{i\theta} + e^{-i\theta}\right)} \end{align} [ tanθの取得方法 ]
