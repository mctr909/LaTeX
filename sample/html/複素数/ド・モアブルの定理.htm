[ 定理 ] \begin{align} e^{i n\theta} &= (e^{i\theta})^n \\ \therefore\cos{n\theta} + i\sin{n\theta} &= (\cos{\theta} + i\sin{\theta})^n \end{align}
[ 確認式 ] n=2の場合を確認してみる \begin{align} e^{i 2\theta} &= (e^{i\theta})^2 \\ &=(\cos{\theta} + i\sin{\theta})^2 \\ &= \cos{\theta} \cdot \cos{\theta} + \cos{\theta} \cdot i\sin{\theta} + i\sin{\theta} \cdot \cos{\theta} + i\sin{\theta} \cdot i\sin{\theta} \\ &=\cos^2{\theta} - \sin^2{\theta} + 2i\sin{\theta} \cdot \cos{\theta} \\ &\quad三角関数の2乗の公式を使って変形する \\ &=2\cos^2{\theta} - 1 + 2i\sin{\theta} \cdot \cos{\theta} \end{align}
\begin{align} \cos{2\theta} &= \frac{e^{i2\theta} + e^{-i2\theta}}{2} \\ &= \frac{(\cos{\theta} + i\sin{\theta})^2}{2} + \frac{(\cos{\theta} - i\sin{\theta})^2}{2} \\ &=\frac{\cos{\theta} \cdot \cos{\theta} + \cos{\theta} \cdot i\sin{\theta} + i\sin{\theta} \cdot \cos{\theta} + i\sin{\theta} \cdot i\sin{\theta}}{2} \\ &+\frac{\cos{\theta} \cdot \cos{\theta} - \cos{\theta} \cdot i\sin{\theta} - i\sin{\theta} \cdot \cos{\theta} + i\sin{\theta} \cdot i\sin{\theta}}{2} \\ &=\frac{2\cos^2{\theta} - 2\sin^2{\theta}}{2} \\ &=\cos^2{\theta} - \sin^2{\theta} \\ &=2\cos^2{\theta} - 1 \quad...\cosの倍角の公式が得られる \end{align} \begin{align} \cosの倍角の公式と確認式の実部が同じであることがわかる \end{align}
\begin{align} \sin{2\theta}&=\frac{e^{i2\theta} - e^{-i2\theta}}{2i} \\ &= \frac{(\cos{\theta} + i\sin{\theta})^2}{2i} - \frac{(\cos{\theta} - i\sin{\theta})^2}{2i} \\ &=\frac{\cos{\theta} \cdot \cos{\theta} + \cos{\theta} \cdot i\sin{\theta} + i\sin{\theta} \cdot \cos{\theta} + i\sin{\theta} \cdot i\sin{\theta}}{2i} \\ &-\frac{\cos{\theta} \cdot \cos{\theta} - \cos{\theta} \cdot i\sin{\theta} - i\sin{\theta} \cdot \cos{\theta} + i\sin{\theta} \cdot i\sin{\theta}}{2i} \\ &=\frac{4i\sin{\theta} \cdot \cos{\theta}}{2i} \\ &=2\sin{\theta} \cdot \cos{\theta} \quad...\sinの倍角の公式が得られる \end{align} \begin{align} \sinの倍角の公式と確認式の虚部が同じであることがわかる \end{align}
