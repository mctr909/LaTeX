\begin{align} &e^{i\theta}=\cos \theta +i\sin \theta \\ &e^{-i\theta}=\cos \theta -i\sin \theta \\ \end{align} \begin{align} &e^{i\theta}+e^{-i\theta}=(\cos \theta +i\sin \theta)+(\cos \theta -i\sin \theta) \\ &=\cos \theta +\cos \theta +i\sin \theta -i\sin \theta \\ &=2\cos \theta \\ &\therefore\cos \theta=\frac{e^{i\theta}+e^{-i\theta}}{2} \end{align} \begin{align} &e^{i\theta}-e^{-i\theta}=(\cos \theta +i\sin \theta)-(\cos \theta -i\sin \theta) \\ &=\cos \theta -\cos \theta +i\sin \theta +i\sin \theta \\ &=2i\sin \theta \\ &\therefore\sin \theta=\frac{e^{i\theta}-e^{-i\theta}}{2i} \end{align}
