\begin{align} &e^{i\left( \alpha+\beta \right)} = e^{i\alpha}e^{i\beta} \\ &=\left( \cos{\alpha} + i\sin{\alpha} \right) \left( \cos{\beta} + i\sin{\beta} \right) \\ &=\cos{\alpha} \cdot \cos{\beta} + \cos{\alpha} \cdot i\sin{\beta} + i\sin{\alpha} \cdot \cos{\beta} + i\sin{\alpha} \cdot i\sin{\beta} \\ &=\cos{\alpha} \cdot \cos{\beta} - \sin{\alpha} \cdot \sin{\beta} + i\left( \sin{\alpha} \cdot \cos{\beta} + \cos{\alpha} \cdot \sin{\beta} \right) \end{align} [ 式1 ]

\begin{align} &\frac{e^{i\left( \alpha+\beta \right)} + e^{-i\left( \alpha+\beta \right)}}{2} \quad...オイラーの公式から\cos{\theta}を取得する方法に\alpha+\betaを代入\\ &=\frac{e^{i\alpha}e^{i\beta}}{2} + \frac{e^{-i\alpha}e^{-i\beta}}{2} \\ &=\frac{\left( \cos{\alpha} + i\sin{\alpha} \right) \left( \cos{\beta} + i\sin{\beta} \right) }{2} \\ &+\frac{\left( \cos{\alpha} - i\sin{\alpha} \right) \left( \cos{\beta} - i\sin{\beta} \right) }{2} \\ &=\frac{\cos{\alpha} \cdot \cos{\beta} + \cos{\alpha} \cdot i\sin{\beta} + i\sin{\alpha} \cdot \cos{\beta} + i\sin{\alpha} \cdot i\sin{\beta} }{2} \\ &+\frac{\cos{\alpha} \cdot \cos{\beta} - \cos{\alpha} \cdot i\sin{\beta} - i\sin{\alpha} \cdot \cos{\beta} + i\sin{\alpha} \cdot i\sin{\beta} }{2} \\ &=\cos{\alpha} \cdot \cos{\beta} - \sin{\alpha} \cdot \sin{\beta} \end{align} [ 式2 ]
