\begin{align} &e^{i 2\theta}=\left(e^{i\theta}\right)^2 \\ &=(\cos\theta +i\sin\theta)^2 \\ &=\cos\theta\cdot\cos\theta +cos\theta\cdot i\sin\theta +i\sin\theta\cdot\cos\theta +i\sin\theta\cdot i\sin\theta \\ &=\cos^2\theta -sin^2\theta +2i(\sin\theta\cdot\cos\theta) \\ \end{align} \begin{align} &\cos(2\theta)=\frac{e^{i2\theta}+e^{-i2\theta}}{2} =\frac{(e^{i\theta})^2+(e^{-i\theta})^2}{2} \\ &=\frac{\left(\cos^2\theta -sin^2\theta +2i(\sin\theta\cdot\cos\theta)\right) +\left(\cos^2\theta -sin^2\theta -2i(\sin\theta\cdot\cos\theta)\right)}{2} \\ &=\frac{\cos^2\theta +\cos^2\theta -sin^2\theta -sin^2\theta +2i(\sin\theta\cdot\cos\theta) -2i(\sin\theta\cdot\cos\theta)}{2} \\ &=\cos^2\theta -sin^2\theta \\ \end{align} \begin{align} &\sin(2\theta)=\frac{e^{i2\theta}-e^{-i2\theta}}{2i} =\frac{(e^{i\theta})^2-(e^{-i\theta})^2}{2i} \\ &=\frac{\left(\cos^2\theta -sin^2\theta +2i(\sin\theta\cdot\cos\theta)\right) -\left(\cos^2\theta -sin^2\theta -2i(\sin\theta\cdot\cos\theta)\right)}{2i} \\ &=\frac{\cos^2\theta -\cos^2\theta -sin^2\theta +sin^2\theta +2i(\sin\theta\cdot\cos\theta) +2i(\sin\theta\cdot\cos\theta)}{2i} \\ &=2(\sin\theta\cdot\cos\theta) \\ \end{align}
