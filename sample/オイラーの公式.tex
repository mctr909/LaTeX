\begin{align}
&e^{ix}=\cos x +i\sin x \\
&e^{-ix}=\cos x -i\sin x \\
\end{align}

\begin{align}
&e^{ix}+e^{-ix}=\cos x +\cos x +i\sin x  -i\sin x = 2\cos x \\
&\therefore\cos x=\frac{e^{ix}+e^{-ix}}{2}
\end{align}
    
\begin{align}
&e^{ix}-e^{-ix}=\cos x -\cos x +i\sin x +i\sin x = 2i\sin x \\
&\therefore\sin x=\frac{e^{ix}-e^{-ix}}{2i}
\end{align}
